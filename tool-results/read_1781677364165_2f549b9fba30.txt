   113→# ━━ TOC Template ━━
   114→class TocDocTemplate(SimpleDocTemplate):
   115→    def afterFlowable(self, flowable):
   116→        if hasattr(flowable, 'bookmark_name'):
   117→            level = getattr(flowable, 'bookmark_level', 0)
   118→            text = getattr(flowable, 'bookmark_text', '')
   119→            key = getattr(flowable, 'bookmark_key', '')
   120→            self.notify('TOCEntry', (level, text, self.page, key))
   121→
   122→def add_heading(text, style, level=0):
   123→    key = 'h_%s' % hashlib.md5(text.encode()).hexdigest()[:8]
   124→    p = Paragraph('<a name="%s"/><b>%s</b>' % (key, text), style)
   125→    p.bookmark_name = text
   126→    p.bookmark_level = level
   127→    p.bookmark_text = text
   128→    p.bookmark_key = key
   129→    return p

   131→def make_table(data, col_widths=None):
   132→    """Create a styled table with consistent formatting."""
   133→    if col_widths is None:
   134→        col_widths = [CONTENT_W / len(data[0])] * len(data[0])
   135→    t = Table(data, colWidths=col_widths, hAlign='CENTER')
   136→    style_commands = [
   137→        ('BACKGROUND', (0, 0), (-1, 0), TABLE_HEADER_COLOR),
   138→        ('TEXTCOLOR', (0, 0), (-1, 0), TABLE_HEADER_TEXT),
   139→        ('GRID', (0, 0), (-1, -1), 0.5, TEXT_MUTED),
   140→        ('VALIGN', (0, 0), (-1, -1), 'MIDDLE'),
   141→        ('LEFTPADDING', (0, 0), (-1, -1), 6),
   142→        ('RIGHTPADDING', (0, 0), (-1, -1), 6),
   143→        ('TOPPADDING', (0, 0), (-1, -1), 5),
   144→        ('BOTTOMPADDING', (0, 0), (-1, -1), 5),
   145→    ]
   146→    for i in range(1, len(data)):
   147→        bg = TABLE_ROW_EVEN if i % 2 == 1 else TABLE_ROW_ODD
   148→        style_commands.append(('BACKGROUND', (0, i), (-1, i), bg))
   149→    t.setStyle(TableStyle(style_commands))
   150→    return t
